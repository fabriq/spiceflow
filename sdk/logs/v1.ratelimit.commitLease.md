/**
 * POST /v1/ratelimit.commitLease
 * Method: POST
 * Tags: ratelimit
 */
export async function commitLease(
  body: V1RatelimitCommitLeaseRequestBody,
  options?: { headers?: Record<string, string> }
): Promise<void> {
  const response = await fetch(`${this.baseUrl}/v1/ratelimit.commitLease`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: this.token ? `Bearer ${this.token}` : '',
      ...options?.headers,
    },
    body: JSON.stringify(body),
  });

  if (!response.ok) {
    const errorData = await response.json();
    throw new ExampleError(errorData.detail || 'Failed to commit lease', {
      status: response.status,
      data: errorData,
    });
  }

  return;
}

interface V1RatelimitCommitLeaseRequestBody {
  cost: number;
  lease: string;
}
