async commitLease(request: V1RatelimitCommitLeaseRequestBody): Promise<void> {
  const response = await this.fetch({
    method: 'POST',
    path: '/v1/ratelimit.commitLease',

    throw new ExampleError(error.detail || 'Failed to commit lease', {

    });
  }
}
