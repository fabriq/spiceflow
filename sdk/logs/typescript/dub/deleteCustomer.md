  // DELETE /customers/{id} - Delete a customer
  async deleteCustomer(id: string): Promise<{ id: string }> {
    const response = await this.fetch<{ id: string }>({
      method: 'DELETE',
      path: `/customers/${encodeURIComponent(id)}`,
