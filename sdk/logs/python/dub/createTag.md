from enum import Enum
from typing import Literal, Optional, TypedDict

class TagColor(str, Enum):
    RED = "red"
    YELLOW = "yellow"
    GREEN = "green"
    BLUE = "blue"
    PURPLE = "purple"
    PINK = "pink"
    BROWN = "brown"

class CreateTagRequest(TypedDict, total=False):
    name: str
    color: TagColor

class Tag(TypedDict):
    id: str
    name: str
    color: TagColor

class ExampleClientAsync:
    # ... existing code ...

    # POST /tags - Create a new tag
    # Tags: Tags
    async def create_tag(
        self,
        name: str,
        color: Optional[TagColor] = None,
    ) -> Tag:
        """Create a new tag for the authenticated workspace.
        
        Args:
            name: The name of the tag to create
            color: The color of the tag. If not provided, a random color will be used
        
        Returns:
            The created tag object
        
        Raises:
            ExampleError: If the request fails
        """
        request_body: CreateTagRequest = {"name": name}
        if color is not None:
            request_body["color"] = color

        response = await self.fetch(
            method="POST",
            path="/tags",
            body=request_body,
        )

        if response.status != 201:
            error_data = await response.json()
            raise ExampleError(
                error=error_data.get("error", {}).get("message", "Unknown error"),
                status=response.status,
                data=error_data,
            )

        return await response.json()
